Make DOM great again!

import { el, on, mount, view } from 'redom';

// Define component

const Login = view({
  init () {
    this.el = el('form',
      on({ submit: this.submit }),

      this.email = el('input', { type: 'email' }),
      this.pass = el('input', { type: 'pass' }),
      this.submit = el('button', 'Sign in')
    )
  },
  submit (e) {
    e.preventDefault();

    console.log(this.email.value, this.pass.value);
  }
})

// init "app"

const app = el(Login);

// Mount to DOM

mount(document.body, app);
 
import { el, list, mount } from 'redom';

// Define components

const Cell = view({
  init () {
    this.el = el('td')
  },
  update (data) {
    this.el.textContent = data;
  }
});

const Row = view({
  init () {
    this.el = el('tr',
      this.cols = list(Cell)
    );
  },
  update (data) {
    this.cols.update(data);
  }
});

const Table = view({
  init () {
    this.el = el('table',
      this.rows = list(Row)
    )
  },
  update (data) {
    this.rows.update(data.tbody);
  }
})

// Init the app

const app = el(Table);

// Mount to DOM

mount(document.body, app);

// update app

app.update({
  tbody: [
    [ 1, 2, 3 ]
  ]
});
